    ```javascript
    (() => {
      const notes = $('S3b Fetch Prior Notes').all()
      const now = new Date()
      if (!notes || notes.length === 0) return 'No prior notes available.'

      return 'Prior clinical context (last ' + notes.length + ' notes, most recent first):\n\n' +
        notes.map((n, i) => {
          let parsed
          try {
            parsed = typeof n.json.structured_note === 'string'
              ? JSON.parse(n.json.structured_note)
              : n.json.structured_note
          } catch (e) { return '[' + (i + 1) + '] Unable to parse note' }
          const created = new Date(n.json.created_at)
          const hoursAgo = Math.round((now - created) / 3600000)
          const ageLabel = hoursAgo < 24
            ? hoursAgo + ' hours ago'
            : Math.round(hoursAgo / 24) + ' days ago'
          return '[' + (i + 1) + '] ' + ageLabel + ':\n  Assessment: ' + (parsed.assessment || 'N/A') + '\n  Plan: ' + (parsed.plan || 'N/A')
        }).join('\n\n')
    })()
    ```
